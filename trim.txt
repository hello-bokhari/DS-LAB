import heapq

class UtilityEnvironment:
    def __init__(self):
        self.cost_grid = [
            [1, 1, 5, 5],
            [5, 1, 1, 1],
            [5, 5, 5, 1]
        ]
        self.goal = (2, 3)

    def get_percept(self):
        return self.cost_grid, self.goal

class UtilityAgent:
    def __init__(self, start_pos):
        self.pos = start_pos

    def find_best_utility(self, grid, goal):
        pq = [(0, self.pos, [self.pos])]
        visited = {}

        while pq:
            cost, curr, path = heapq.heappop(pq)

            if curr == goal:
                return path, cost

            if curr in visited and visited[curr] <= cost:
                continue
            visited[curr] = cost

            y, x = curr
            for dy, dx in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
                ny, nx = y + dy, x + dx
                if 0 <= ny < len(grid) and 0 <= nx < len(grid[0]):
                    new_cost = cost + grid[ny][nx]
                    heapq.heappush(pq, (new_cost, (ny, nx), path + [(ny, nx)]))
        return None

u_env = UtilityEnvironment()
grid, goal = u_env.get_percept()
u_agent = UtilityAgent((0, 0))
path, total_cost = u_agent.find_best_utility(grid, goal)
print(f"Utility Agent Best Path: {path} | Total Cost: {total_cost}")

from collections import deque

class Environment:
    def __init__(self):
        self.grid = [
            [0, 0, 0, 1],
            [1, 1, 0, 1],
            [0, 0, 0, 0]
        ]
        self.goal = (2, 3)

    def get_percept(self):
        return self.grid, self.goal

class GoalAgent:
    def __init__(self, start_pos):
        self.pos = start_pos

    def plan_path(self, grid, goal):
        queue = deque([(self.pos, [self.pos])])
        visited = {self.pos}
        
        while queue:
            (curr_y, curr_x), path = queue.popleft()
            
            if (curr_y, curr_x) == goal:
                return path
            
            for dy, dx in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
                ny, nx = curr_y + dy, curr_x + dx
                if 0 <= ny < len(grid) and 0 <= nx < len(grid[0]) and grid[ny][nx] == 0:
                    if (ny, nx) not in visited:
                        visited.add((ny, nx))
                        queue.append(((ny, nx), path + [(ny, nx)]))
        return None

from collections import deque

def bfs(tree, start, goal):
    visited = set()
    queue = deque([(start, [start])])
    visited.add(start)

    while queue:
        (node, path) = queue.popleft()
        
        if node == goal:
            return path

        for neighbor in tree.get(node, []):
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append((neighbor, path + [neighbor]))
                
    return None

import random
class LearningAgent:
    def __init__(self):
        self.performance_standard = 22
        self.knowledge = {}
        
    def act(self, current_temp):
        if current_temp in self.knowledge:
            return self.knowledge[current_temp]
        return random.choice(["Heat", "Cool"])

    def learn(self, state, action, result_temp):
        if abs(result_temp - self.performance_standard) < 2:
            self.knowledge[state] = action
            print(f"Learned: At {state}, {action} is good.")

#Utility-Based Agent
class UtilityBasedAgent:
    def __init__(self):
        self.goal = "Home"
        self.options = {
            "Work": [("Bus", 10, 50), ("Uber", 2, 200)] # (Time, Cost)
        }

    def utility_function(self, time, cost):
        return (time * 0.7) + (cost * 0.3)

    def act(self, location):
        choices = self.options.get(location, [])
        best_action = None
        best_utility = float('inf')

        for name, time, cost in choices:
            u = self.utility_function(time, cost)
            if u < best_utility:
                best_utility = u
                best_action = name
        return f"Take the {best_action}"

agent = UtilityBasedAgent()
print(f"Utility Action: {agent.act('Work')}")

#Goal-Based Agent
class GoalBasedAgent:
    def __init__(self, goal_city):
        self.goal = goal_city
        self.map = {"S": ["A", "B"], "A": ["Goal"], "B": ["Goal"]}

    def act(self, current_city):
        if current_city == self.goal:
            return "Goal Reached!"
        
        neighbors = self.map.get(current_city, [])
        for neighbor in neighbors:
            if neighbor == self.goal or "Goal" in self.map.get(neighbor, []):
                return f"Move to {neighbor}"
        return "Search"

agent = GoalBasedAgent("Goal")
print(f"Goal Action: {agent.act('S')}")

#Model-Based Reflex Agent
class ModelBasedAgent:
    def __init__(self):
        self.model = {"A": "Unknown", "B": "Unknown"} # Internal memory
        self.rules = {"Dirty": "Suck", "Clean": "Move to other room"}

    def update_model(self, location, status):
        self.model[location] = status

    def act(self, location, status):
        self.update_model(location, status)
        
        if status == "Dirty":
            return "Suck"
        
        other_room = "B" if location == "A" else "A"
        if self.model[other_room] == "Clean":
            return "Done/Idle"
        return f"Move to {other_room}"

agent = ModelBasedAgent()
print(f"Model Action: {agent.act('A', 'Clean')}")

import random

def heuristic(state):
    conflicts = 0
    n = len(state)
  
    for i in range(n):
        for j in range(i + 1, n):
            if state[i] == state[j]:
                conflicts += 1
            elif abs(state[i] - state[j]) == abs(i - j):
                conflicts += 1    
    return conflicts


def hill_climbing(n):
    current = [random.randint(0, n-1) for _ in range(n)]
    
    while True:
        current_h = heuristic(current)
        
        if current_h == 0:
            return current
        
        neighbors = []

        for col in range(n):
            for row in range(n):
                if row != current[col]:
                    neighbor = current.copy()
                    neighbor[col] = row
                    neighbors.append(neighbor)

        next_state = None
        next_h = current_h
        
        for state in neighbors:
            h = heuristic(state)
            if h < next_h:
                next_h = h
                next_state = state
        
        if next_state is None:
            return current
        current = next_state

n = 4
solution = hill_climbing(n)

print("Solution:", solution)
print("Conflicts:", heuristic(solution))

def hill_climbing(graph, start, goal, heuristic):
    current = start
    path = [current]
    
    while True:
        if current == goal:
            return path
        
        neighbors = graph.get(current, [])
        
        if not neighbors:
            return path
        
        best_neighbor = min(neighbors, key=lambda x: heuristic[x[0]])[0]
        best_h = heuristic[best_neighbor]
        
        if best_h >= heuristic[current]:
            return path  
        
        current = best_neighbor
        path.append(current)

#beam_search
def beam_search(graph, start, goal, beam_width):
    queue = [[start]]
    
    while queue:
        all_candidates = []
        
        for path in queue:
            node = path[-1]

            if node == goal:
                return path

            for (neighbor, h_value) in graph.get(node, []):
                new_path = list(path)
                new_path.append(neighbor)
                all_candidates.append((new_path, h_value))

        all_candidates.sort(key=lambda x: x[1])
        queue = [path for (path, h) in all_candidates[:beam_width]]
    return None

import heapq

def heuristic(a, b):
    return abs(a[0] - b[0]) + abs(a[1] - b[1])

def greedy_bfs_maze(maze, start, goal):
    rows, cols = len(maze), len(maze[0])

    open_list = []
    heapq.heappush(open_list, (heuristic(start, goal), start))

    came_from = {}
    visited = set()

    while open_list:
        _, current = heapq.heappop(open_list)

        if current == goal:
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            return path[::-1]

        visited.add(current)

        neighbors = [
            (current[0] + 1, current[1]),
            (current[0] - 1, current[1]),
            (current[0], current[1] + 1),
            (current[0], current[1] - 1)
        ]

        for r, c in neighbors:
            if 0 <= r < rows and 0 <= c < cols and maze[r][c] == 0:
                neighbor = (r, c)

                if neighbor not in visited:
                    heapq.heappush(open_list, (heuristic(neighbor, goal), neighbor))
                    if neighbor not in came_from:
                        came_from[neighbor] = current
    return None

def heuristic(a, b):
    return abs(a[0] - b[0]) + abs(a[1] - b[1])

def astar(maze, start, goal):
    rows, cols = len(maze), len(maze[0])

    open_list = []
    heapq.heappush(open_list, (0, start))

    came_from = {}
    g_cost = {start: 0}

    while open_list:
        _, current = heapq.heappop(open_list)

        if current == goal:
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            return path[::-1]
        
        neighbors = [
            (current[0] + 1, current[1]),
            (current[0] - 1, current[1]),
            (current[0], current[1] + 1),
            (current[0], current[1] - 1)
        ]

        for neighbor in neighbors:
            r, c = neighbor

            if 0 <= r < rows and 0 <= c < cols and maze[r][c] == 0:
                tentative_g = g_cost[current] + 1

                if neighbor not in g_cost or tentative_g < g_cost[neighbor]:
                    g_cost[neighbor] = tentative_g
                    f_cost = tentative_g + heuristic(neighbor, goal)

                    heapq.heappush(open_list, (f_cost, neighbor))
                    came_from[neighbor] = current

    return None

#astar_graph
import heapq

def astar(graph, heuristic, start, goal):
    openlist = []
    heapq.heappush(openlist,(0,start))

    cameFrom = {}
    costSoFar = {start:0}

    while openlist:
        currentcost, currentnode = heapq.heappop(openlist)

        if currentnode == goal:
            break

        for neighbor, cost in graph[currentnode]:
            newcost = costSoFar[currentnode] + cost

            if neighbor not in costSoFar or newcost < costSoFar[neighbor]:
                costSoFar[neighbor] = newcost
                priority = newcost + heuristic[neighbor]
                heapq.heappush(openlist, (priority,neighbor))
                cameFrom[neighbor] = currentnode

    path = []
    node = goal
    if goal not in cameFrom and start != goal:
        print("Goal not found")
        return
    
    while node != start:
        path.append(node)
        node = cameFrom[node]
    path.append(start)
    path.reverse()

    return path, costSoFar[goal]

graph = {
    'S': [('A', 1), ('B', 4)],
    'A': [('C', 2)],
    'B': [('C', 1)],
    'C': [('G', 3)],
    'G': []
}

heuristic = {
    'S': 5,
    'A': 4,
    'B': 2,
    'C': 1,
    'G': 0
}

path, cost = astar(graph, heuristic, 'S', 'G')
print("Path:", path)
print("Cost:", cost)

#genetic_algorithim
import random

n = 4
population_size = 20
generations = 100

def fitness(chromosome):
    # fitneess = max_pairs - attacking_pairs
    attacking_pairs = 0
    max_pairs = n * (n-1) // 2

    for i in range(n):
        for j in range(i+1, n):
            if chromosome[i] == chromosome[j]:
                attacking_pairs += 1
            if abs(chromosome[i]-chromosome[j]) == abs(i-j):
                attacking_pairs += 1
    
    return max_pairs - attacking_pairs

def crateChromosone():
    return [random.randint(0, n-1) for _ in range(n)]

def createPopulation():
    return [crateChromosone() for _ in range(population_size)]

def selection(population):
    population.sort(key=lambda x: fitness(x), reverse=True)
    return population[:len(population)//2]

def crossover(parent1, parent2):
    n = len(parent1)
    point = random.randint(0, n-2)
    child = parent1[:point] + parent2[point:]
    return child

def mutation(chromosome):
    n = len(chromosome)

    if random.random() < 0.1:
        index = random.randint(0, n-1)
        chromosome[index] = random.randint(0, n-1)
    return chromosome

def genetic_algorithim():
    population = createPopulation()
    for generation in range(generations):
        population = selection(population)

        newPopulation = population.copy()

        while len(newPopulation) < population_size:
            parent1, parent2 = random.sample(population, 2)
            child = crossover(parent1,parent2)
            child = mutation(child)

            newPopulation.append(child)
        population = newPopulation

        #check best solution
        best = max(population, key=lambda x: fitness(x))

        if fitness(best) == n*(n-1) // 2:
            print("solution found at generation", generation)
            print("Board:", best)
            return best
        
        print("best found:", best)
        return best

genetic_algorithim()
