#Optimization Problem (ILP)
from ortools.sat.python import cp_model

def main():
    model = cp_model.CpModel()

    var_upper_bound = max(50, 45, 37)

    x = model.NewIntVar(0, var_upper_bound, "x")
    y = model.NewIntVar(0, var_upper_bound, "y")
    z = model.NewIntVar(0, var_upper_bound, "z")

    # Constraints
    model.Add(2 * x + 7 * y + 3 * z <= 50)
    model.Add(3 * x - 5 * y + 7 * z <= 45)
    model.Add(5 * x + 2 * y - 6 * z <= 37)

    # Objective
    model.Maximize(2 * x + 2 * y + 3 * z)

    solver = cp_model.CpSolver()
    status = solver.Solve(model)

    if status == cp_model.OPTIMAL or status == cp_model.FEASIBLE:
        print("Maximum value =", solver.ObjectiveValue())
        print("x =", solver.Value(x))
        print("y =", solver.Value(y))
        print("z =", solver.Value(z))
    else:
        print("No solution found.")

    print("\nStatistics")
    print("Status:", solver.StatusName(status))
    print("Conflicts:", solver.NumConflicts())
    print("Branches:", solver.NumBranches())
    print("Wall time:", solver.WallTime())

main()

#Minimax Algorithm
import math


# Node class
class Node:
    def __init__(self, value=None):
        self.value = value
        self.children = []
        self.minmax_value = None


# Minimax Agent
class MinimaxAgent:
    def __init__(self, depth):
        self.depth = depth

    def formulate_goal(self, node):
        # Goal is achieved when root minimax value is computed
        return "Goal reached" if node.minmax_value is not None else "Searching"

    def act(self, node, environment):
        goal_status = self.formulate_goal(node)

        if goal_status == "Goal reached":
            return f"Minimax value for root node: {node.minmax_value}"
        else:
            return environment.compute_minimax(node, self.depth)


# Environment
class Environment:
    def __init__(self, tree):
        self.tree = tree
        self.computed_nodes = []

    def get_percept(self, node):
        return node

    def compute_minimax(self, node, depth, maximizing_player=True):
        # Base case
        if depth == 0 or not node.children:
            self.computed_nodes.append(node.value)
            return node.value

        # Maximizer
        if maximizing_player:
            value = -math.inf

            for child in node.children:
                child_value = self.compute_minimax(
                    child,
                    depth - 1,
                    False
                )
                value = max(value, child_value)

            node.minmax_value = value
            self.computed_nodes.append(node.value)
            return value

        # Minimizer
        else:
            value = math.inf

            for child in node.children:
                child_value = self.compute_minimax(
                    child,
                    depth - 1,
                    True
                )
                value = min(value, child_value)

            node.minmax_value = value
            self.computed_nodes.append(node.value)
            return value


# Run Agent
def run_agent(agent, environment, start_node):
    percept = environment.get_percept(start_node)
    agent.act(percept, environment)

import math


# Node class
class Node:
    def __init__(self, value=None):
        self.value = value
        self.children = []
        self.minmax_value = None


# Minimax Agent with Alpha-Beta
class MinimaxAgent:
    def __init__(self, depth):
        self.depth = depth

    def formulate_goal(self, node):
        return "Goal reached" if node.minmax_value is not None else "Searching"

    def act(self, node, environment):
        goal_status = self.formulate_goal(node)

        if goal_status == "Goal reached":
            return f"Minimax value for root node: {node.minmax_value}"
        else:
            return environment.alpha_beta_search(
                node,
                self.depth,
                -math.inf,
                math.inf,
                True
            )


# Environment
class Environment:
    def __init__(self, tree):
        self.tree = tree
        self.computed_nodes = []

    def get_percept(self, node):
        return node

    def alpha_beta_search(
        self,
        node,
        depth,
        alpha,
        beta,
        maximizing_player=True
    ):
        self.computed_nodes.append(node.value)

        # Base case
        if depth == 0 or not node.children:
            return node.value

        # Maximizer
        if maximizing_player:
            value = -math.inf

            for child in node.children:
                value = max(
                    value,
                    self.alpha_beta_search(
                        child,
                        depth - 1,
                        alpha,
                        beta,
                        False
                    )
                )

                alpha = max(alpha, value)

                # Pruning
                if beta <= alpha:
                    print("Pruned node:", child.value)
                    break

            node.minmax_value = value
            return value

        # Minimizer
        else:
            value = math.inf

            for child in node.children:
                value = min(
                    value,
                    self.alpha_beta_search(
                        child,
                        depth - 1,
                        alpha,
                        beta,
                        True
                    )
                )

                beta = min(beta, value)

                # Pruning
                if beta <= alpha:
                    print("Pruned node:", child.value)
                    break

            node.minmax_value = value
            return value


# Run Agent
def run_agent(agent, environment, start_node):
    percept = environment.get_percept(start_node)
    agent.act(percept, environment)

import math


def alpha_beta(node, depth, alpha, beta, maximizing):
    
    # Leaf node
    if depth == 0 or isinstance(node, int):
        return node

    # MAX player
    if maximizing:
        value = -math.inf

        # Sort descending for maximum pruning
        node = sorted(node, reverse=True)

        for child in node:
            value = max(
                value,
                alpha_beta(child, depth - 1, alpha, beta, False)
            )

            alpha = max(alpha, value)

            if beta <= alpha:
                print("Pruned at MAX")
                break

        return value

    # MIN player
    else:
        value = math.inf

        # Sort ascending for maximum pruning
        node = sorted(node)

        for child in node:
            value = min(
                value,
                alpha_beta(child, depth - 1, alpha, beta, True)
            )

            beta = min(beta, value)

            if beta <= alpha:
                print("Pruned at MIN")
                break

        return value


def Alpha_Beta_Defense(tree):

    # Dynamically sort main attack branches
    # Best ordering for pruning
    sorted_tree = sorted(
        tree,
        key=lambda branch: max(branch),
        reverse=True
    )

    result = alpha_beta(
        sorted_tree,
        2,
        -math.inf,
        math.inf,
        True
    )

    return result

import math


board = [' ' for _ in range(9)]


def print_board():
    for i in range(0, 9, 3):
        print(board[i], "|", board[i+1], "|", board[i+2])


def is_winner(player):
    win_positions = [
        [0,1,2],
        [3,4,5],
        [6,7,8],
        [0,3,6],
        [1,4,7],
        [2,5,8],
        [0,4,8],
        [2,4,6]
    ]

    for pos in win_positions:
        if all(board[i] == player for i in pos):
            return True

    return False


def is_draw():
    return ' ' not in board


def minimax(depth, alpha, beta, maximizing):

    if is_winner('O'):
        return 1

    if is_winner('X'):
        return -1

    if is_draw():
        return 0

    if maximizing:
        best = -math.inf

        for i in range(9):
            if board[i] == ' ':
                board[i] = 'O'
                score = minimax(depth + 1, alpha, beta, False)
                board[i] = ' '

                best = max(best, score)
                alpha = max(alpha, best)

                if beta <= alpha:
                    break

        return best

    else:
        best = math.inf

        for i in range(9):
            if board[i] == ' ':
                board[i] = 'X'
                score = minimax(depth + 1, alpha, beta, True)
                board[i] = ' '

                best = min(best, score)
                beta = min(beta, best)

                if beta <= alpha:
                    break

        return best


def best_move():
    best_score = -math.inf
    move = -1

    for i in range(9):
        if board[i] == ' ':
            board[i] = 'O'
            score = minimax(0, -math.inf, math.inf, False)
            board[i] = ' '

            if score > best_score:
                best_score = score
                move = i

    return move


def play_game():

    while True:
        print_board()

        player_move = int(input("Enter position (0-8): "))

        if board[player_move] == ' ':
            board[player_move] = 'X'
        else:
            print("Invalid move")
            continue

        if is_winner('X'):
            print_board()
            print("You win!")
            break

        if is_draw():
            print_board()
            print("Draw!")
            break

        ai_move = best_move()
        board[ai_move] = 'O'

        print("AI played:", ai_move)

        if is_winner('O'):
            print_board()
            print("AI wins!")
            break

        if is_draw():
            print_board()
            print("Draw!")
            break

#Basic Markov Model (Marble Problem)
import numpy as np


states = ["Red", "Blue"]

transition_matrix = np.array([
    [0.5, 0.5],
    [0.5, 0.5]
])


def simulate_markov_process(initial_state, num_steps):
    current_state = initial_state
    state_sequence = [current_state]

    for _ in range(num_steps):

        if current_state == "Red":
            next_state = np.random.choice(
                states,
                p=transition_matrix[0]
            )
        else:
            next_state = np.random.choice(
                states,
                p=transition_matrix[1]
            )

        state_sequence.append(next_state)
        current_state = next_state

    return state_sequence


initial_state = "Red"
num_steps = 10

sequence = simulate_markov_process(
    initial_state,
    num_steps
)

print(" -> ".join(sequence))

#Weather Markov Model
import numpy as np


states = ["Sunny", "Cloudy", "Rainy"]

transition_matrix = np.array([
    [0.6, 0.3, 0.1],  
    [0.3, 0.4, 0.3],  
    [0.2, 0.3, 0.5]   
])


def simulate_weather(initial_state, days):
    current_state = initial_state
    weather_sequence = [current_state]

    for _ in range(days - 1):

        index = states.index(current_state)

        next_state = np.random.choice(
            states,
            p=transition_matrix[index]
        )

        weather_sequence.append(next_state)
        current_state = next_state

    return weather_sequence

    print("At least 3 rainy days: Yes")

    # Chase
    [0.2, 0.6, 0.2],

    # Search
    [0.3, 0.3, 0.4]
])

    current_state = initial_state
    sequence = [current_state]

    for _ in range(steps):

        next_state = np.random.choice(

        sequence.append(next_state)

    return sequence

wcss = []

for i in range(1, 11):
    kmeans = KMeans(
        n_clusters=i,
        init='k-means++',
        random_state=42
    )
    kmeans.fit(X)
    wcss.append(kmeans.inertia_)

plt.xlabel("Number of Clusters")

plt.show()

# Feature Scaling

X_scaled = scaler.fit_transform(X)

# Train KMeans

    n_clusters=5,
    init='k-means++',
    random_state=42
)

y_pred = kmeans.fit_predict(X_scaled)

class UtilityAgent:
    def __init__(self, start_pos):
        self.pos = start_pos

    def find_best_utility(self, grid, goal):
        pq = [(0, self.pos, [self.pos])]
        visited = {}

        while pq:
            cost, curr, path = heapq.heappop(pq)

            if curr == goal:
                return path, cost

            if curr in visited and visited[curr] <= cost:
                continue
            visited[curr] = cost

            y, x = curr
            for dy, dx in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
                ny, nx = y + dy, x + dx
                if 0 <= ny < len(grid) and 0 <= nx < len(grid[0]):
                    new_cost = cost + grid[ny][nx]
                    heapq.heappush(pq, (new_cost, (ny, nx), path + [(ny, nx)]))
        return None

def bfs(tree, start, goal):
    visited = set()
    queue = deque([(start, [start])])
    visited.add(start)

    while queue:
        (node, path) = queue.popleft()
        
        if node == goal:
            return path

        for neighbor in tree.get(node, []):
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append((neighbor, path + [neighbor]))
                
    return None

import random

def heuristic(state):
    conflicts = 0
    n = len(state)
  
    for i in range(n):
        for j in range(i + 1, n):
            if state[i] == state[j]:
                conflicts += 1
            elif abs(state[i] - state[j]) == abs(i - j):
                conflicts += 1    
    return conflicts


def hill_climbing(n):
    current = [random.randint(0, n-1) for _ in range(n)]
    
    while True:
        current_h = heuristic(current)
        
        if current_h == 0:
            return current
        
        neighbors = []

        for col in range(n):
            for row in range(n):
                if row != current[col]:
                    neighbor = current.copy()
                    neighbor[col] = row
                    neighbors.append(neighbor)

        next_state = None
        next_h = current_h
        
        for state in neighbors:
            h = heuristic(state)
            if h < next_h:
                next_h = h
                next_state = state
        
        if next_state is None:
            return current
        current = next_state

def hill_climbing(graph, start, goal, heuristic):
    current = start
    path = [current]
    
    while True:
        if current == goal:
            return path
        
        neighbors = graph.get(current, [])
        
        if not neighbors:
            return path
        
        best_neighbor = min(neighbors, key=lambda x: heuristic[x[0]])[0]
        best_h = heuristic[best_neighbor]
        
        if best_h >= heuristic[current]:
            return path  
        
        current = best_neighbor
        path.append(current)

def beam_search(graph, start, goal, beam_width):
    queue = [[start]]
    
    while queue:
        all_candidates = []
        
        for path in queue:
            node = path[-1]

            if node == goal:
                return path

            for (neighbor, h_value) in graph.get(node, []):
                new_path = list(path)
                new_path.append(neighbor)
                all_candidates.append((new_path, h_value))

        all_candidates.sort(key=lambda x: x[1])
        queue = [path for (path, h) in all_candidates[:beam_width]]
    return None

import heapq

def astar(graph, heuristic, start, goal):
    openlist = []
    heapq.heappush(openlist,(0,start))

    cameFrom = {}
    costSoFar = {start:0}

    while openlist:
        currentcost, currentnode = heapq.heappop(openlist)

        if currentnode == goal:
            break

        for neighbor, cost in graph[currentnode]:
            newcost = costSoFar[currentnode] + cost

            if neighbor not in costSoFar or newcost < costSoFar[neighbor]:
                costSoFar[neighbor] = newcost
                priority = newcost + heuristic[neighbor]
                heapq.heappush(openlist, (priority,neighbor))
                cameFrom[neighbor] = currentnode

    path = []
    node = goal
    if goal not in cameFrom and start != goal:
        print("Goal not found")
        return
    
    while node != start:
        path.append(node)
        node = cameFrom[node]
    path.append(start)
    path.reverse()

    return path, costSoFar[goal]

import random

n = 4
population_size = 20
generations = 100

def fitness(chromosome):
    # fitneess = max_pairs - attacking_pairs
    attacking_pairs = 0
    max_pairs = n * (n-1) // 2

    for i in range(n):
        for j in range(i+1, n):
            if chromosome[i] == chromosome[j]:
                attacking_pairs += 1
            if abs(chromosome[i]-chromosome[j]) == abs(i-j):
                attacking_pairs += 1
    
    return max_pairs - attacking_pairs

def crateChromosone():
    return [random.randint(0, n-1) for _ in range(n)]

def createPopulation():
    return [crateChromosone() for _ in range(population_size)]

def selection(population):
    population.sort(key=lambda x: fitness(x), reverse=True)
    return population[:len(population)//2]

def crossover(parent1, parent2):
    n = len(parent1)
    point = random.randint(0, n-2)
    child = parent1[:point] + parent2[point:]
    return child

def mutation(chromosome):
    n = len(chromosome)

    if random.random() < 0.1:
        index = random.randint(0, n-1)
        chromosome[index] = random.randint(0, n-1)
    return chromosome

def genetic_algorithim():
    population = createPopulation()
    for generation in range(generations):
        population = selection(population)

        newPopulation = population.copy()

        while len(newPopulation) < population_size:
            parent1, parent2 = random.sample(population, 2)
            child = crossover(parent1,parent2)
            child = mutation(child)

            newPopulation.append(child)
        population = newPopulation

        #check best solution
        best = max(population, key=lambda x: fitness(x))

        if fitness(best) == n*(n-1) // 2:
            print("solution found at generation", generation)
            print("Board:", best)
            return best
        
        print("best found:", best)
        return best
